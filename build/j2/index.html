Let's write a simple function to calculate the Gini Impurity. We will use this as a metric for determining the structure of our decsion tree. The Gini Impurity (in the context of the Titanic dataset) is a measure of the likelihood that a selected passenger's survival status was mislabeled, assuming that that passenger's assigned status was chosen probabilistically based on the distribution of survivals within the passenger set.

In [7]:
def gini(survived, total):
    survival_rate = survived/total
    death_rate = 1 - survival_rate
    prob_survival = survival_rate
    prob_death = death_rate
    false_pos = prob_death * survival_rate
    false_neg = prob_survival * death_rate
    gini = false_pos + false_neg
    return gini

def test_split(survived1, split1, survived2, split2):
    initial_gini = gini(survived1 + survived2, split1 + split2)
    split1_gini = gini(survived1, split1)
    split2_gini = gini(survived2, split2)
    weight1 = split1/(split1 + split2)
    weight2 = split2/(split1 + split2)
    weighted_split_gini = (split1_gini * weight1) + (split2_gini * weight2)
    diff = weighted_split_gini - initial_gini
    return diff

cv = KFold(n_splits=10)            # Desired number of Cross Validation folds
accuracies = list()
max_attributes = len(list(test))
depth_range = range(1, max_attributes + 1)

# Testing max_depths from 1 to max attributes
# Uncomment prints for details about each Cross Validation pass
for depth in depth_range:
    fold_accuracy = []
    tree_model = tree.DecisionTreeClassifier(max_depth = depth)
    # print("Current max depth: ", depth, "\n")
    for train_fold, valid_fold in cv.split(train):
        f_train = train.loc[train_fold] # Extract train data with cv indices
        f_valid = train.loc[valid_fold] # Extract valid data with cv indices

        model = tree_model.fit(X = f_train.drop(['Survived'], axis=1), 
                               y = f_train["Survived"]) # We fit the model with the fold train data
        valid_acc = model.score(X = f_valid.drop(['Survived'], axis=1), 
                                y = f_valid["Survived"])# We calculate accuracy with the fold validation data
        fold_accuracy.append(valid_acc)

    avg = sum(fold_accuracy)/len(fold_accuracy)
    accuracies.append(avg)
    # print("Accuracy per fold: ", fold_accuracy, "\n")
